描述 ​

实现一个简单的虚拟DOM库，类似于React的基本功能。它提供了创建虚拟DOM元素(createElement)和将虚拟DOM渲染到实际DOM中(render)的能力。

方法 ​

  1. createElement函数：接受元素类型(type)、属性(props)和子元素(children)作为参数，返回一个虚拟DOM元素。这个虚拟DOM元素是一个对象，包含type、props和children属性。

  2. render函数：接受一个虚拟DOM元素和一个容器DOM元素。它将虚拟DOM转换为实际的DOM结构，并将其附加到容器元素中。这个过程包括：

    • 如果虚拟DOM元素是字符串，创建一个文本节点。
    • 如果是元素对象，创建相应类型的DOM元素，然后递归地将其子元素也渲染到DOM中。
    • 处理虚拟DOM元素的属性，将它们应用到创建的DOM元素上。

代码 ​

ts
type Props = {
    [key: string]: any;
}

interface Element {
    type: string;
    props: Props | null;
    children: (Element | string)[];
}

/**
 * 创建一个虚拟 DOM 元素
 * @param type 元素类型
 * @param props 元素属性
 * @param children 子元素
 * @returns 虚拟 DOM 元素
 */
function createElement(type: string, props: Props | null,
 ...children: (Element | string)[]): Element {
    return {
        type,
        props,
        children
    }
}

/**
 * 将虚拟 DOM 元素渲染到指定的容器中
 * @param element 虚拟 DOM 元素
 * @param container 容器元素
 */
function render(element: Element, container: HTMLElement): void {
    let dom: HTMLElement;
    
    if (typeof element === 'string') {
        dom = document.createTextNode(element);
    } else {
        dom = document.createElement(element.type);
        if (element.props) {
            Object.keys(element.props).forEach(key => {
                const value = element.props![key];
                if (key in dom) {
                    (dom as any)[key] = value;
                } else {
                    dom.setAttribute(key, value);
                }
            });
        }

        element.children.forEach(child => render(child, dom));
    }

    container.appendChild(dom);
}

// 示例用法
const element = createElement('div', { id: 'container' },
    createElement('h1', null, 'Hello'),
    createElement('p', null, 'This is a paragraph.')
);

// 渲染到容器中
const container = document.getElementById('app')!;
if (container) {
    render(element, container);
}
